 20 -def upload( filename ) : 

 21      """ 
 22      Upload a track to The Echo Nest's analyzer for analysis and later 
 23      retrieval of track information. 
 24   
 25      :param filename: Either the path to a local file or a valid URL. 
 26   
 27      :return: (id, info) 
 28      """ 
 29      if config.API_KEY.startswith('Replace'): 
 30          print "Please edit src/echonest/web/config.py to use your Echo Nest API key.\n" 
 31          sys.exit(-1) 
 32      fields = { "api_key" : config.API_KEY, 
 33                 "wait" : "Y" } 
 34      method = 'upload' 
 35      if os.path.isfile( filename ) :  
 36          response = util.postChunked( host = config.API_HOST,  
 37                                       selector = config.API_SELECTOR + method, 
 38                                       fields = fields.items(),  
 39                                       files = (( 'file', open(filename, 'r')), ) 
 40                                       ) 
 41      else : 
 42          print "it's a url" 
 43          # Assume the filename is a URL. 
 44          fields['url'] = filename 
 45          params = urllib.urlencode(fields) 
 46          url = 'http://%s%s%s?%s' % (config.API_HOST, config.API_SELECTOR, method, params) 
 47          f = urllib.urlopen( url ) 
 48          response = f.read() 
 49      return util.parseXMLString(response)
